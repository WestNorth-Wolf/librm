127 LkMotor(hal::CanInterface &can, usize id, bool reversed = false)
128 : CanDevice(can, 0x140 + id), id_{id}, reversed_{reversed} {}

134 template <LkMotorType t = type, std::enable_if_t<t == LkMotorType::kMS, int> = 0>
135 void SetPower(f32 power) {
136 if (reversed_) {
137 power = -power;
138 }
139 i16 power_cmd = modules::Clamp(power, -850.f, 850.f);
140 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
141 tx_buffer_[0] = Instruction::kOpenLoopControl;
142 tx_buffer_[4] = *(u8 *)(&power_cmd);
143 tx_buffer_[5] = *((u8 *)(&power_cmd) + 1);
144 can_->Write(0x140 + id_, tx_buffer_, 8);
145 }
146
151 template <LkMotorType t = type, std::enable_if_t<t != LkMotorType::kMS, int> = 0>
152 void SetTorque(f32 torque_nm) {
153 if (reversed_) {
154 torque_nm = -torque_nm;
155 }
156 i16 torque_cmd;
157 if constexpr (type == LkMotorType::kMF) {
158 torque_nm = modules::Clamp(torque_nm, -16.5f, 16.5f);
159 torque_cmd = modules::Map(torque_nm, -16.5f, 16.5f, -2048, 2048);
160 } else if constexpr (type == LkMotorType::kMG) {
161 torque_nm = modules::Clamp(torque_nm, -33.f, 33.f);
162 torque_cmd = modules::Map(torque_nm, -33.f, 33.f, -2048, 2048);
163 }
164 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
165 tx_buffer_[0] = Instruction::kTorqueControl;
166 tx_buffer_[4] = *(u8 *)(&torque_cmd);
167 tx_buffer_[5] = *((u8 *)(&torque_cmd) + 1);
168 can_->Write(0x140 + id_, tx_buffer_, 8);
169 }
170
175 void SetSpeed(f32 speed_rad_ps) {
176 if (reversed_) {
177 speed_rad_ps = -speed_rad_ps;
178 }
179 const float dps_100x = speed_rad_ps * 180.f / M_PI * 100.f; // rad/s -> dps, 0.01dps/LSB
180 const i32 power_cmd = dps_100x; // 上位机里还有一个Max Speed值限制，这个值如果大于上位机里设置的值，会被电机限制掉
181 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
182 tx_buffer_[0] = Instruction::kSpeedControl;
183 tx_buffer_[4] = *(u8 *)(&power_cmd);
184 tx_buffer_[5] = *((u8 *)(&power_cmd) + 1);
185 tx_buffer_[6] = *((u8 *)(&power_cmd) + 2);
186 tx_buffer_[7] = *((u8 *)(&power_cmd) + 3);
187 can_->Write(0x140 + id_, tx_buffer_, 8);
188 }

196 void SendInstruction(Instruction instruction) {
197 switch (instruction) {
198 case kEnable: {
199 feedback_.enabled = true;
200 break;
201 }
202 case kDisable: {
203 feedback_.enabled = false;
204 break;
205 }
206 default: {
207 break;
208 }
209 }
210 std::memset(&tx_buffer_[1], 0, sizeof(tx_buffer_) - 1);
211 tx_buffer_[0] = instruction;
212 can_->Write(0x140 + id_, tx_buffer_, 8);
213 }

221 tx_buffer_[1] = param_type;

223 can_->Write(0x140 + id_, tx_buffer_, 8);

249 private:
254 void RxCallback(const hal::CanFrame *msg) override {
255 ReportStatus(kOk);
256 switch (msg->data[0]) {
257 case Instruction::kOpenLoopControl:
258 case Instruction::kSpeedControl: {
259 feedback_.temperature = *(i8 *)(&msg->data[1]);
260 feedback_.power = (msg->data[2] | (msg->data[3] << 8));
261 feedback_.speed_rad_ps = (i16)(msg->data[4] | (msg->data[5] << 8)) / 180.f * M_PI;
262 const u16 encoder_raw = (msg->data[6] | (msg->data[7] << 8));
263 if constexpr (encoder_bits == 15) {
264 feedback_.position_rad = modules::Map(encoder_raw, 0, 32767, 0.f, 2.f * M_PI);
265 } else if constexpr (encoder_bits == 18) {
266 feedback_.position_rad = modules::Map(encoder_raw, 0, 65535, 0.f, 2.f * M_PI);
267 }
268 break;
269 }
270 case Instruction::kTorqueControl: {
271 feedback_.temperature = *(i8 *)(&msg->data[1]);
272 const i16 iq_raw = (msg->data[2] | (msg->data[3] << 8));
273 if constexpr (type == LkMotorType::kMF) {
274 feedback_.iq = modules::Map(iq_raw, -2048, 2048, -16.5f, 16.5f);
275 } else if constexpr (type == LkMotorType::kMG) {
276 feedback_.iq = modules::Map(iq_raw, -2048, 2048, -33.f, 33.f);
277 }
278 feedback_.speed_rad_ps = (i16)(msg->data[4] | (msg->data[5] << 8)) / 180.f * M_PI;
279 const u16 encoder_raw = (msg->data[6] | (msg->data[7] << 8));
280 if constexpr (encoder_bits == 15) {
281 feedback_.position_rad = modules::Map(encoder_raw, 0, 32767, 0.f, 2.f * M_PI);
282 } else if constexpr (encoder_bits == 18) {
283 feedback_.position_rad = modules::Map(encoder_raw, 0, 65535, 0.f, 2.f * M_PI);
284 }
285 break;
286 }
287 case Instruction::kReadState1: {
288 feedback_.temperature = msg->data[1];
289 const u16 voltage_raw = (msg->data[3] | (msg->data[4] << 8));
290 feedback_.voltage = voltage_raw / 10.f;
291 *(u8 *)(&feedback_.error_state) = msg->data[7];
292 break;
293 }
294 case Instruction::kReadControlParam: {
295 switch (msg->data[1]) {
296 case ControlParamType::kPositionPid: {
297 feedback_.control_param.position_kp = (msg->data[2] | (msg->data[3] << 8));
298 feedback_.control_param.position_ki = (msg->data[4] | (msg->data[5] << 8));
299 feedback_.control_param.position_kd = (msg->data[6] | (msg->data[7] << 8));
300 break;
301 }
302 case ControlParamType::kSpeedPid: {
303 feedback_.control_param.speed_kp = (msg->data[2] | (msg->data[3] << 8));
304 feedback_.control_param.speed_ki = (msg->data[4] | (msg->data[5] << 8));
305 feedback_.control_param.speed_kd = (msg->data[6] | (msg->data[7] << 8));
306 break;
307 }
308 case ControlParamType::kCurrentPid: {
309 feedback_.control_param.current_kp = (msg->data[2] | (msg->data[3] << 8));
310 feedback_.control_param.current_ki = (msg->data[4] | (msg->data[5] << 8));
311 feedback_.control_param.current_kd = (msg->data[6] | (msg->data[7] << 8));
312 }
313 default: {
314 break;
315 }
316 }
317 break;
318 }
319 case Instruction::kReadPositionMulti: {
320 int64_t pos_multi_raw = ( //
321 msg->data[1] | (msg->data[2] << 8) | //
322 (msg->data[3] << 16) | (msg->data[4] << 24) | //
323 (msg->data[5] << 32) | (msg->data[6] << 40) | //
324 (msg->data[7] << 48) //
325 );
326 feedback_.position_multi_rad = (pos_multi_raw / 100.f) / 180.f * M_PI;
327 break;
328 }
329 // TODO: 其他反馈报文，暂时用不到
330 default: {
331 break;
332 }
333 }
334 }
