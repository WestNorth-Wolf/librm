42template <typename Tp>
43static constexpr Tp max_(std::initializer_list<Tp> il) {
44 if (il.size() == 0) {
45 return 0;
46 }
47 Tp max = *il.begin();
48 for (auto &i : il) {
49 if (i > max) {
50 max = i;
51 }
52 }
53 return max;
54}

105 auto Forward(f32 vx, f32 vy, f32 w) {
106 if (vx == 0.f && vy == 0.f && w == 0.f) { // 理论上不应该直接比较浮点数，但是两个同类型浮点字面量比较应该没问题
107 forward_result_.lf_steer_position = M_PI / 4.f;
108 forward_result_.rf_steer_position = 3 * M_PI / 4.f;
109 forward_result_.lr_steer_position = 3 * M_PI / 4.f;
110 forward_result_.rr_steer_position = M_PI / 4.f;
111 } else {
112 forward_result_.lf_steer_position =
113 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
114 forward_result_.rf_steer_position =
115 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
116 forward_result_.lr_steer_position =
117 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
118 forward_result_.rr_steer_position =
119 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
120 }
121
122 forward_result_.lf_wheel_speed =
123 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
124 forward_result_.rf_wheel_speed =
125 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
126 forward_result_.lr_wheel_speed =
127 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
128 forward_result_.rr_wheel_speed =
129 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
130
131 return forward_result_;
132 }

144 auto Forward(f32 vx, f32 vy, f32 w, f32 current_lf_angle, f32 current_rf_angle, f32 current_lr_angle,
145 f32 current_rr_angle) {
146 Forward(vx, vy, w);
147
148 // 依次计算每个舵的目标角度和目前角度的差值，如果差值大于90度，就把目标舵角加180度，轮速取反。
149 // 这样可以保证舵角变化量始终小于90度，加快舵的响应速度
150 if (std::abs(Wrap(current_lf_angle - forward_result_.lf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
151 forward_result_.lf_steer_position = Wrap(forward_result_.lf_steer_position + M_PI, 0, 2 * M_PI);
152 forward_result_.lf_wheel_speed = -forward_result_.lf_wheel_speed;
153 }
154 if (std::abs(Wrap(current_rf_angle - forward_result_.rf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
155 forward_result_.rf_steer_position = Wrap(forward_result_.rf_steer_position + M_PI, 0, 2 * M_PI);
156 forward_result_.rf_wheel_speed = -forward_result_.rf_wheel_speed;
157 }
158 if (std::abs(Wrap(current_lr_angle - forward_result_.lr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
159 forward_result_.lr_steer_position = Wrap(forward_result_.lr_steer_position + M_PI, 0, 2 * M_PI);
160 forward_result_.lr_wheel_speed = -forward_result_.lr_wheel_speed;
161 }
162 if (std::abs(Wrap(current_rr_angle - forward_result_.rr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
163 forward_result_.rr_steer_position = Wrap(forward_result_.rr_steer_position + M_PI, 0, 2 * M_PI);
164 forward_result_.rr_wheel_speed = -forward_result_.rr_wheel_speed;
165 }
166
167 return forward_result_;
168 }

174 } forward_result_{};

192 rear_x_ = base_len / 2.0f;

206 auto Forward(f32 vx, f32 vy, f32 w) {
207 using modules::IsNear;
208 if (IsNear(vx, 0.f, 1e-6f) && //
209 IsNear(vy, 0.f, 1e-6f) && //
210 IsNear(w, 0.f, 1e-6f)) {
211 // 三个舵摆成一个圆周阵型（各轮切线方向），这样可以随时开始小陀螺旋转
212 forward_result_.front_steer_position = 0.f; // 前轮朝右
213 forward_result_.lr_steer_position = atan2f(rear_x_, rear_y_); // 左后轮朝左前
214 forward_result_.rr_steer_position = atan2f(-rear_x_, rear_y_); // 右后轮朝左后
215 } else {
216 // 前轮位于 (0, chassis_radius_)
217 const f32 front_vx = vx - w * chassis_radius_;
218 const f32 front_vy = vy;
219 forward_result_.front_steer_position = atan2f(front_vy, front_vx);
220
221 // 左后轮位于 (-rear_x_, rear_y_)
222 const f32 lr_vx = vx - w * rear_y_;
223 const f32 lr_vy = vy - w * rear_x_;
224 forward_result_.lr_steer_position = atan2f(lr_vy, lr_vx);
225
226 // 右后轮位于 (rear_x_, rear_y_)
227 const f32 rr_vx = vx - w * rear_y_;
228 const f32 rr_vy = vy + w * rear_x_;
229 forward_result_.rr_steer_position = atan2f(rr_vy, rr_vx);
230 }
231
232 // 计算轮速（速度向量的模）
233 const f32 front_vx = vx - w * chassis_radius_;
234 const f32 front_vy = vy;
235 forward_result_.front_wheel_speed = sqrtf(front_vx * front_vx + front_vy * front_vy);
236
237 const f32 lr_vx = vx - w * rear_y_;
238 const f32 lr_vy = vy - w * rear_x_;
239 forward_result_.lr_wheel_speed = sqrtf(lr_vx * lr_vx + lr_vy * lr_vy);
240
241 const f32 rr_vx = vx - w * rear_y_;
242 const f32 rr_vy = vy + w * rear_x_;
243 forward_result_.rr_wheel_speed = sqrtf(rr_vx * rr_vx + rr_vy * rr_vy);
244
245 return forward_result_;
246 }

303 auto Forward(f32 vx, f32 vy, f32 wz, bool normalize = false) {
304 forward_result_.lf_speed = vx + vy + wz;
305 forward_result_.rf_speed = vx - vy + wz;
306 forward_result_.lr_speed = -vx + vy + wz;
307 forward_result_.rr_speed = -vx - vy + wz;
308 if (normalize) {
309 const f32 max_speed = max_({std::abs(forward_result_.lf_speed), std::abs(forward_result_.rf_speed),
310 std::abs(forward_result_.lr_speed), std::abs(forward_result_.rr_speed)});
311 if (max_speed > 1) {
312 forward_result_.lf_speed /= max_speed;
313 forward_result_.rf_speed /= max_speed;
314 forward_result_.lr_speed /= max_speed;
315 forward_result_.rr_speed /= max_speed;
316 }
317 }
318
319 return forward_result_;
320 }

328 auto Inverse(f32 lf_speed, f32 rf_speed, f32 lr_speed, f32 rr_speed) {
329 inverse_result_.vx = (lf_speed + rf_speed - lr_speed - rr_speed) / 4;
330 inverse_result_.vy = (-lf_speed + rf_speed - lr_speed + rr_speed) / 4;
331 inverse_result_.wz = (lf_speed + rf_speed + lr_speed + rr_speed) / 4;
332 // normalize
333 f32 max_speed = max_({std::abs(inverse_result_.vx), std::abs(inverse_result_.vy), std::abs(inverse_result_.wz)});
334 if (max_speed > 1) {
335 inverse_result_.vx /= max_speed;
336 inverse_result_.vy /= max_speed;
337 inverse_result_.wz /= max_speed;
338 }
339
340 return inverse_result_;
341 }
342 auto forward_result() const { return forward_result_; }
343 auto inverse_result() const { return inverse_result_; }
