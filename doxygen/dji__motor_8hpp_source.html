96 static void SendCommand(hal::CanInterface &can) {
97 auto target_can = tx_buf_.find(&can);
98 if (target_can == tx_buf_.end()) {
99 return;
100 }
101 auto &buffers = target_can->second;
102
103 if (buffers.dirty_200) {
104 can.Write(0x200, buffers.data_200, 8);
105 buffers.dirty_200 = false;
106 }
107 if (buffers.dirty_1ff) {
108 can.Write(0x1ff, buffers.data_1ff, 8);
109 buffers.dirty_1ff = false;
110 }
111 if (buffers.dirty_2ff) {

183 constexpr static i16 GetCurrentBound() {
184 switch (motor_type) {
185 case DjiMotorType::kGM6020:

187 case DjiMotorType::kM3508:
188 return 16384;
189 case DjiMotorType::kM2006:

209 DjiMotor(hal::CanInterface &can, u16 id, bool reversed = false)
210 : CanDevice(can, GetRxIdBase() + id), id_(id), reversed_(reversed) {
211 // 检查id是否合法
212 if (id_ < 1 || id_ > 8) {
213 rm::Throw(std::runtime_error("DjiMotor: invalid motor id"));
214 }
215 if constexpr (motor_type == DjiMotorType::kGM6020) {
216 if (id_ > 7) { // 6020没有8号
217 rm::Throw(std::runtime_error("DjiMotor: invalid motor id for GM6020"));
218 }
219 }
220 // 如果这个电机所属的CAN总线还没有对应的发送缓冲区，就创建一个
221 if (tx_buf_.find(&can) == tx_buf_.end()) {
222 tx_buf_.insert({&can, TxBuffers{}});
223 }
224 }

231 void SetCurrent(i16 current) {
232 // 限幅到电机能接受的最大电流
233 current = modules::Clamp(current, -GetCurrentBound(), GetCurrentBound());
234 // 处理反转
235 if (this->reversed_) {
236 current = -current;
237 }
238 // 找到这个电机所属的CAN总线的发送缓冲区
239 auto buffers = tx_buf_.find(this->can_);
240 if (buffers == tx_buf_.end()) {
241 // 这个电机没有对应的CAN总线缓冲区，理论上不会发生，但是为安全起见还是加上这个判断
242 return;
243 }
244 auto &[_, buf] = *buffers;
245 // 根据电机型号和ID(this->id_)找到对应的发送缓冲区，写入电流值，标记缓冲区已修改
246 if constexpr (motor_type == DjiMotorType::kGM6020) {
247 if (1 <= this->id_ && this->id_ <= 4) {
248 buf.data_1ff[(this->id_ - 1) * 2] = (current >> 8) & 0xff;
249 buf.data_1ff[(this->id_ - 1) * 2 + 1] = current & 0xff;
250 buf.dirty_1ff = true;
251 } else if (5 <= this->id_ && this->id_ <= 8) {
252 buf.data_2ff[(this->id_ - 5) * 2] = (current >> 8) & 0xff;
253 buf.data_2ff[(this->id_ - 5) * 2 + 1] = current & 0xff;
254 buf.dirty_2ff = true;
255 }
256 } else if constexpr (motor_type == DjiMotorType::kM3508 || //
257 motor_type == DjiMotorType::kM2006) {
258 if (1 <= this->id_ && this->id_ <= 4) {
259 buf.data_200[(this->id_ - 1) * 2] = (current >> 8) & 0xff;
260 buf.data_200[(this->id_ - 1) * 2 + 1] = current & 0xff;
261 buf.dirty_200 = true;
262 } else if (5 <= this->id_ && this->id_ <= 8) {
263 buf.data_1ff[(this->id_ - 5) * 2] = (current >> 8) & 0xff;
264 buf.data_1ff[(this->id_ - 5) * 2 + 1] = current & 0xff;
265 buf.dirty_1ff = true;
266 }
267 } else if constexpr (motor_type == DjiMotorType::kGM6020Current) {
268 if (1 <= this->id_ && this->id_ <= 4) {
269 buf.data_1fe[(this->id_ - 1) * 2] = (current >> 8) & 0xff;
270 buf.data_1fe[(this->id_ - 1) * 2 + 1] = current & 0xff;
271 buf.dirty_1fe = true;
272 } else if (5 <= this->id_ && this->id_ <= 8) {
273 buf.data_2fe[(this->id_ - 5) * 2] = (current >> 8) & 0xff;
274 buf.data_2fe[(this->id_ - 5) * 2 + 1] = current & 0xff;
275 buf.dirty_2fe = true;
276 }
277 }
278 }

281 [[nodiscard]] u16 encoder() const { return this->encoder_; }

287 [[nodiscard]] f32 pos_rad() const { return (f32)this->encoder() / kMaxEncoderValue * M_PI * 2; }

290 private:
291 void RxCallback(const hal::CanFrame *msg) override {
292 ReportStatus(kOk);
293 this->encoder_ = (msg->data[0] << 8) | msg->data[1];
294 this->rpm_ = (msg->data[2] << 8) | msg->data[3];
295 this->current_ = (msg->data[4] << 8) | msg->data[5];
296 this->temperature_ = msg->data[6];
297 if(reversed_){
298 this->encoder = this->kMaxEncoderValue - this->encoder;
299 this->rpm_ = -this->rpm_;
300 this->current = -this->current_;
301 }
302 }

304 u16 id_{};
305 bool reversed_{};
307 u16 encoder_{};
308 i16 rpm_{};
309 i16 current_{};
310 u8 temperature_{};
