80 void Send(const uint8_t *data, uint32_t size) {
81 // 更新最后发送时间，数据帧本身就能维持连接
82 last_tx_time_ = std::chrono::steady_clock::now();
83
84 // 1. 生成一个随机的帧ID，用于标识本次传输的所有分片
85 tx_buf_.frame_start.frame_id = static_cast<uint8_t>(rng_() & 0xff);
86
87 // 2. 准备起始帧，包含元数据：帧ID、总载荷长度和CRC8校验值
88 tx_buf_.frame_start.payload_len = size;
89 tx_buf_.frame_start.payload_crc8 = modules::Crc8(data, size, modules::CRC8_INIT);
90 // 发送起始帧
91 can_->Write(tx_stdid_base_ + StdIdOffset::kFrameStart, reinterpret_cast<const uint8_t *>(&tx_buf_.frame_start),
92 sizeof(tx_buf_.frame_start));
93
94 // 3. 循环发送数据分片
95 uint16_t fragment_seq = 0;
96 uint32_t bytes_sent = 0;
97 while (bytes_sent < size) {
98 tx_buf_.payload.frame_id = tx_buf_.frame_start.frame_id;
99 tx_buf_.payload.fragment_seq = fragment_seq++;
100 // 计算当前分片的数据大小
101 uint8_t fragment_size =
102 std::min(static_cast<uint32_t>(sizeof(tx_buf_.payload.fragment_data)), size - bytes_sent);
103 tx_buf_.payload.fragment_len = fragment_size;
104 std::memcpy(tx_buf_.payload.fragment_data, data + bytes_sent, fragment_size);
105 // 发送数据分片，DLC固定为64
106 can_->Write(tx_stdid_base_ + StdIdOffset::kPayload, reinterpret_cast<const uint8_t *>(&tx_buf_.payload), 64);
107 bytes_sent += fragment_size;
108 }
109
110 // 4. 发送结束帧，标识数据传输完成
111 tx_buf_.frame_end.frame_id = tx_buf_.frame_start.frame_id;
112 can_->Write(tx_stdid_base_ + StdIdOffset::kFrameEnd, reinterpret_cast<const uint8_t *>(&tx_buf_.frame_end),
113 sizeof(tx_buf_.frame_end));
114 }

122 void SendHeartbeatIfNeeded() {
123 auto now = std::chrono::steady_clock::now();
124 auto elapsed_ms = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_tx_time_).count();
125 if (elapsed_ms >= kHeartbeatIntervalMs) {
126 can_->Write(tx_stdid_base_ + StdIdOffset::kHeartbeat, nullptr, 0);
127 last_tx_time_ = now;

134 void RxCallback(const hal::CanFrame *msg) override {
135 // 收到任何帧都更新连接状态，因为任何帧都能证明对方在线
136 ReportStatus(kOk);
137
138 // 判断帧类型
139 if (msg->rx_std_id == rx_stdid_base_ + StdIdOffset::kHeartbeat) {
140 // 心跳帧，已在上面更新状态
141 } else if (msg->rx_std_id == rx_stdid_base_ + StdIdOffset::kFrameStart) {
142 // 1. 收到起始帧，开始一个新的数据帧接收会话
143 const auto *frame_start = reinterpret_cast<const decltype(tx_buf_.frame_start) *>(msg->data.data());
144 if (msg->dlc != sizeof(*frame_start)) {
145 return; // 帧长度错误，直接丢弃
146 }
147 if (frame_start->payload_len > kMaxPayloadBytes) {
148 // 载荷超出单帧允许的最大长度，直接丢弃该会话
149 rx_sessions_.erase(frame_start->frame_id);
150 return;
151 }
152 // 检查会话池是否已满，如果满了且是新会话，则淘汰最不活跃的会话
153 if (rx_sessions_.find(frame_start->frame_id) == rx_sessions_.end() && rx_sessions_.full()) {
154 EvictLeastRecentlyUsedSession();
155 }
156 auto &session = rx_sessions_[frame_start->frame_id];
157 session.payload_len = frame_start->payload_len;
158 session.payload_crc8 = frame_start->payload_crc8;
159 session.next_fragment_seq = 0;
160 session.last_active_time = std::chrono::steady_clock::now(); // 更新活跃时间
161 session.buffer.clear();
162 session.buffer.reserve(frame_start->payload_len);
163 } else if (msg->rx_std_id == rx_stdid_base_ + StdIdOffset::kPayload) {
164 // 2. 收到数据分片
165 const auto *payload = reinterpret_cast<const decltype(tx_buf_.payload) *>(msg->data.data());
166 auto session_it = rx_sessions_.find(payload->frame_id);
167 if (session_it == rx_sessions_.end()) {
168 return; // 未找到对应会话，说明起始帧缺失
169 }
170 auto &session = session_it->second;
171 // 校验分片序号确保顺序正确
172 if (payload->fragment_seq != session.next_fragment_seq) {
173 rx_sessions_.erase(session_it); // 顺序错误，丢弃该会话
174 return;
175 }
176 const uint32_t data_len = payload->fragment_len;
177 if (session.buffer.size() + data_len > session.payload_len) {
178 rx_sessions_.erase(session_it); // 数据超长，丢弃该会话
179 return;
180 }
181 session.buffer.insert(session.buffer.end(), payload->fragment_data, payload->fragment_data + data_len);
182 session.next_fragment_seq++;
183 session.last_active_time = std::chrono::steady_clock::now(); // 更新活跃时间
184
185 } else if (msg->rx_std_id == rx_stdid_base_ + StdIdOffset::kFrameEnd) {
186 // 3. 收到结束帧
187 const auto *frame_end = reinterpret_cast<const decltype(tx_buf_.frame_end) *>(msg->data.data());
188 auto session_it = rx_sessions_.find(frame_end->frame_id);
189 if (session_it == rx_sessions_.end()) {
190 return; // 没有对应的会话，忽略
191 }
192 auto &session = session_it->second;
193 if (session.buffer.size() != session.payload_len) {
194 rx_sessions_.erase(session_it); // 长度不一致，放弃接收
195 return;
196 }
197 // 4. 数据重组和CRC校验
198 uint8_t calculated_crc = modules::Crc8(session.buffer.empty() ? nullptr : session.buffer.data(),
199 session.buffer.size(), modules::CRC8_INIT);
200 if (calculated_crc == session.payload_crc8) {
201 // 5. CRC校验通过，调用回调函数处理完整的数据帧
202 if (frame_recv_callback_) {
203 std::string received_data(session.buffer.begin(), session.buffer.end());
204 frame_recv_callback_(received_data);
205 }
206 }
207 rx_sessions_.erase(session_it);
208 }
209 }

264 void EvictLeastRecentlyUsedSession() {
265 if (rx_sessions_.empty()) return;
266
267 auto oldest_it = rx_sessions_.begin();
268 auto oldest_time = oldest_it->second.last_active_time;
269
270 for (auto it = rx_sessions_.begin(); it != rx_sessions_.end(); ++it) {
271 if (it->second.last_active_time < oldest_time) {
272 oldest_time = it->second.last_active_time;
273 oldest_it = it;
274 }
275 }
276
277 rx_sessions_.erase(oldest_it);
278 }
