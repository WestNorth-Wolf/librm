166template <size_t MaxDevices, bool UseStdFunctionCallback = true, size_t MaxSummaryStringLength = 512>
167class DeviceManager {
168 using CallbackType = //
169 std::conditional_t<UseStdFunctionCallback, //
170 std::function<void(Device *)>, //
171 etl::delegate<void(Device *)>>;
172
173 public:
174 DeviceManager() = default;
175
176 DeviceManager(std::initializer_list<Device *> devices) {
177 if (devices.size() <= MaxDevices) {
178 for (auto *device : devices) {
179 devices_.push_back({device, Device::kUnknown}); // 初始状态为未知
180 }
181 } else {
182 Throw(std::runtime_error("DeviceManager: Too many devices!"));
183 }
184 }
185
186 DeviceManager &operator<<(Device *device) {
187 if (devices_.size() < MaxDevices) {
188 devices_.push_back({device, Device::kUnknown}); // 初始状态为未知
189 } else {
190 Throw(std::runtime_error("DeviceManager: Too many devices!"));
191 }
192 return *this;
193 }
194
199 bool Update() {
200 all_device_ok_ = true;
201
202 // 清空状态列表
203 offline_devices_.clear();
204 fault_devices_.clear();
205 ok_devices_.clear();
206 unknown_devices_.clear();
207
208 for (auto &entry : devices_) {
209 const Device::Status current_status = entry.device->online_status();
210 if (current_status != Device::kOk) {
211 all_device_ok_ = false;
212 }
213
214 // 根据状态添加到对应列表
215 switch (current_status) {
216 case Device::kOffline:
217 offline_devices_.push_back(entry.device);
218 break;
219 case Device::kFault:
220 fault_devices_.push_back(entry.device);
221 break;
222 case Device::kOk:
223 ok_devices_.push_back(entry.device);
224 break;
225 case Device::kUnknown:
226 unknown_devices_.push_back(entry.device);
227 break;
228 }
229
230 // 设备状态发生变化，触发回调
231 if (entry.prev_status != current_status) {
232 for (const auto &callback : status_change_callbacks_) {
233 callback(entry.device);
234 }
235 }
236 entry.prev_status = current_status;
237 }
238 return all_device_ok_;
239 }

246 void OnDeviceStatusChange(CallbackType callback) {
247 if (status_change_callbacks_.size() < 10) {
248 status_change_callbacks_.emplace_back(callback);
249 } else {
250 Throw(std::runtime_error("DeviceManager: Too many status change callbacks!"));
251 }
252 }

281 etl::string<MaxSummaryStringLength> GetSummaryString() const {
282 constexpr size_t kTruncationMarkLength = 3; // "..." 的长度
283 etl::string<MaxSummaryStringLength> summary;
284
285 if (all_device_ok_) {
286 return "All devices OK.";
287 }
288
289 // 辅助函数1：安全地追加字符串，如果空间不足则截断并返回 false
290 auto try_append = [&](const char *str) -> bool {
291 const size_t str_len = strlen(str);
292 // etl::string::available() -> 剩余容量 < 要追加的字符串长度 + 截断标记"..."长度？
293 if (summary.available() < str_len + kTruncationMarkLength) {
294 summary += "...";
295 return false; // 空间不足，截断
296 }
297 summary += str;
298 return true; // 追加成功
299 };
300
301 // 辅助函数2：追加一类设备列表（"Offline: dev1, dev2; "）
302 auto append_device_category = [&](const etl::vector<Device *, MaxDevices> &devices,
303 const char *category_name) -> bool {
304 if (devices.empty()) {
305 return true; // 这个类别没有设备，继续下一个
306 }
307
308 // 类别名称（"Offline: "）
309 if (!try_append(category_name)) {
310 return false;
311 }
312 if (!try_append(": ")) {
313 return false;
314 }
315
316 // 设备名称列表
317 for (size_t i = 0; i < devices.size(); ++i) {
318 if (!try_append(devices[i]->name().c_str())) {
319 return false;
320 }
321 if (i < devices.size() - 1) {
322 if (!try_append(", ")) {
323 return false;
324 }
325 }
326 }
327
328 // 追加类别结束符
329 return try_append("; ");
330 };
331
332 // 按顺序处理各个类别，任何一个类别截断都会终止后续处理
333 if (!append_device_category(offline_devices_, "Offline")) return summary;
334 if (!append_device_category(fault_devices_, "Fault")) return summary;
335 if (!append_device_category(unknown_devices_, "Unknown")) return summary;
336
337 return summary;
338 }
