95 PowerInfo CalculatePower(const MotorState& state) const {
96 PowerInfo info;
97
98 // 计算输出转矩: τ = KT * I_torque
99 f32 torque_current = state.give_current * params_.current_to_torque_current;
100 info.torque = params_.kt * torque_current;
101
102 // 计算机械功率: P_mech = τ·ω / 9.55
103 info.mechanical_power = (info.torque * state.speed_rpm) * params_.speed_to_mechanical_power;
104
105 // 计算损耗功率: P_loss = k1·ω² + k2·I² + C
106 f32 speed_squared = state.speed_rpm * state.speed_rpm;
107 f32 current_squared = state.give_current * state.give_current;
108 info.loss_power = params_.k1 * speed_squared + params_.k2 * current_squared + params_.constant;
109
110 // 总功率
111 info.total_power = info.mechanical_power + info.loss_power;
112
113 return info;
114 }

123 f32 CalculateCtrlForPower(f32 target_power, f32 speed_rpm, bool is_positive_direction) const {
124 // 求解二次方程: a·I² + b·I + c = 0
125 // 其中功率模型为: P = (KT·I_torque)·ω/9.55 + k1·ω² + k2·I² + C
126 // P = TORQUE_COEFFICIENT·I·ω + k1·ω² + k2·I² + C
127
128 const f32 a = params_.k2;
129 const f32 b = params_.torque_coeff * speed_rpm;
130 const f32 c = params_.k1 * speed_rpm * speed_rpm - target_power + params_.constant;
131
132 // 判别式
133 const f32 discriminant = b * b - 4.0f * a * c;
134
135 if (discriminant < 0) {
136 // 无实数解，目标功率无法达到
137 return 0.0f;
138 }
139
140 const f32 sqrt_discriminant = std::sqrt(discriminant);
141 f32 current;
142
143 // 根据转矩方向选择解
144 if (is_positive_direction) {
145 current = (-b + sqrt_discriminant) / (2.0f * a);
146 } else {
147 current = (-b - sqrt_discriminant) / (2.0f * a);
148 }
149
150 // 限幅
151 if (current > params_.max_current) {
152 current = params_.max_current;
153 } else if (current < -params_.max_current) {
154 current = -params_.max_current;
155 }
156
157 return current;
158 }

167 template <usize NMotors>
168 void DistributePower(const std::array<MotorState, NMotors> motor_states, const f32* initial_currents,
169 f32 max_total_power, f32* output_currents) const {
170 // 计算初始总功率
171 f32 initial_total_power = 0.0f;
172 f32 initial_powers[NMotors];
173
174 for (size_t i = 0; i < NMotors; ++i) {
175 MotorState state = motor_states[i];
176 state.give_current = initial_currents[i];
177
178 const PowerInfo info = CalculatePower(state);
179 initial_powers[i] = info.total_power;
180
181 // 负功率不计入（过渡状态）
182 if (initial_powers[i] > 0) {
183 initial_total_power += initial_powers[i];
184 }
185 }
186
187 // 如果超过最大功率，按比例缩放
188 if (initial_total_power > max_total_power) {
189 const f32 power_scale = max_total_power / initial_total_power;
190
191 for (size_t i = 0; i < NMotors; ++i) {
192 const f32 scaled_power = initial_powers[i] * power_scale;
193
194 if (scaled_power < 0) {
195 output_currents[i] = initial_currents[i];
196 continue;
197 }
198
199 // 根据缩放后的功率重新计算电流
200 const bool is_positive = initial_currents[i] > 0;
201 output_currents[i] = CalculateCtrlForPower(scaled_power, motor_states[i].speed_rpm, is_positive);
202 }
203 } else {
204 // 不超过功率限制，直接使用初始电流
205 for (size_t i = 0; i < NMotors; ++i) {
206 output_currents[i] = initial_currents[i];
207 }
208 }
209 }

229 }
