51 f32 real_position() const { return rotation_count_ * encoder_cycle_ + last_encoder_; }

57 void Update(f32 encoder) {
58 if (first_update_) {
59 last_encoder_ = encoder;
60 first_update_ = false;
61 return;
62 }
63
64 const f32 delta = encoder - last_encoder_;
65
66 if (delta > encoder_cycle_ / 2.0f) {
67 // 编码器从大值跳到小值，向下跨越（逆时针转动穿过 0 点）
68 rotation_count_--;
69 } else if (delta < -encoder_cycle_ / 2.0f) {
70 // 编码器从小值跳到大值，向上跨越（顺时针转动穿过 0 点）
71 rotation_count_++;
72 }
73
74 last_encoder_ = encoder;
75 }
76
80 void Reset() {
81 last_encoder_ = 0.0f;
82 rotation_count_ = 0;
83 first_update_ = true;
84 }
85
86 private:
87 f32 encoder_cycle_;
88 f32 last_encoder_{0.0f};
89 int rotation_count_{0};
90 bool first_update_{true};
91};
