165 void Update(u32 keycode) {
166 u32 regular_mask = ~modifier_mask_;
167
168 u32 current_modifiers = keycode & modifier_mask_;
169 u32 current_regular_keys = keycode & regular_mask;
170
171 u32 prev_regular_keys = prev_keycode_ & regular_mask;
172
173 // 检测新按下的普通键
174 u32 newly_pressed_regular = current_regular_keys & ~prev_regular_keys;
175 // 检测新释放的普通键
176 u32 newly_released_regular = prev_regular_keys & ~current_regular_keys;
177
178 // 处理新按下的普通键
179 for (u32 bit = 0; bit < 32; ++bit) {
180 u32 key_mask = 1U << bit;
181
182 // 跳过modifier键
183 if (key_mask & modifier_mask_) {
184 continue;
185 }
186
187 if (newly_pressed_regular & key_mask) {
188 // 普通键是新按下的，使用当前modifier状态来判断组合键
189 HandleKeyDown(key_mask, current_modifiers);
190 }
191 }
192
193 // 处理新释放的普通键
194 for (u32 bit = 0; bit < 32; ++bit) {
195 u32 key_mask = 1U << bit;
196
197 if (key_mask & modifier_mask_) {
198 continue;
199 }
200
201 if (newly_released_regular & key_mask) {
202 // 释放键时，使用该键按下时记录的modifier状态
203 HandleKeyUp(key_mask);
204 }
205 }
206
207 // 更新按键按下时的modifier状态记录
208 UpdatePressedKeyModifiers(newly_pressed_regular, current_modifiers);
209 ClearReleasedKeyModifiers(newly_released_regular);
210
211 prev_keycode_ = keycode;
212 }

257 static void SortBindings(BindingVector& bindings) {
258 etl::sort(bindings.begin(), bindings.end(), [](const KeyBinding& a, const KeyBinding& b) {
259 return CountBits(a.required_modifiers) > CountBits(b.required_modifiers);
260 });
261 }
262
266 static u32 CountBits(u32 n) {
267 u32 count = 0;
268 while (n) {
269 count += n & 1;
270 n >>= 1;
271 }
272 return count;
273 }

280 void HandleKeyDown(u32 key, u32 current_modifiers) {
281 bool matched = false;
282
283 // 查找匹配的绑定（已按modifier数量降序排序）
284 for (auto& binding : key_down_bindings_) {
285 if (binding.key != key) {
286 continue;
287 }
288
289 // 检查modifier是否精确匹配
290 // 要求的modifier必须全部按下，且不能有额外的modifier
291 if (binding.required_modifiers == current_modifiers) {
292 // 精确匹配
293 KeyEvent event{key, current_modifiers, KeyEvent::Type::kKeyDown};
294 binding.callback(event);
295 matched = true;
296 break; // 只触发第一个匹配的绑定（最具体的）
297 }
298 }
299
300 // 如果没有精确匹配，并且当前有modifier键按下，
301 // 则不触发无modifier的单键回调（模仿桌面OS行为）
302 // 如果没有modifier键按下，且有无modifier的绑定，则触发它
303 if (!matched && current_modifiers == 0) {
304 for (auto& binding : key_down_bindings_) {
305 if (binding.key == key && binding.required_modifiers == 0) {
306 KeyEvent event{key, 0, KeyEvent::Type::kKeyDown};
307 binding.callback(event);
308 break;
309 }
310 }
311 }
312 }
313
318 void HandleKeyUp(u32 key) {
319 // 查找该键按下时的modifier状态
320 u32 modifiers_when_pressed = 0;
321 for (const auto& state : pressed_key_modifiers_) {
322 if (state.key == key) {
323 modifiers_when_pressed = state.modifiers;
324 break;
325 }
326 }
327
328 bool matched = false;
329
330 // 查找匹配的绑定
331 for (auto& binding : key_up_bindings_) {
332 if (binding.key != key) {
333 continue;
334 }
335
336 if (binding.required_modifiers == modifiers_when_pressed) {
337 KeyEvent event{key, modifiers_when_pressed, KeyEvent::Type::kKeyUp};
338 binding.callback(event);
339 matched = true;
340 break;
341 }
342 }
343
344 if (!matched && modifiers_when_pressed == 0) {
345 for (auto& binding : key_up_bindings_) {
346 if (binding.key == key && binding.required_modifiers == 0) {
347 KeyEvent event{key, 0, KeyEvent::Type::kKeyUp};
348 binding.callback(event);
349 break;
350 }
351 }
352 }
353 }

358 void UpdatePressedKeyModifiers(u32 newly_pressed, u32 modifiers) {
359 for (u32 bit = 0; bit < 32; ++bit) {
360 u32 key_mask = 1U << bit;
361 if ((newly_pressed & key_mask) && !pressed_key_modifiers_.full()) {
362 pressed_key_modifiers_.push_back({key_mask, modifiers});
363 }
364 }
365 }
366
370 void ClearReleasedKeyModifiers(u32 released) {
371 auto it = pressed_key_modifiers_.begin();
372 while (it != pressed_key_modifiers_.end()) {
373 if (it->key & released) {
374 it = pressed_key_modifiers_.erase(it);
375 } else {
376 ++it;
377 }
378 }
379 }
380
381 u32 modifier_mask_;
382 u32 prev_keycode_{0};
383 BindingVector key_down_bindings_;
384 BindingVector key_up_bindings_;
385 PressedKeyVector pressed_key_modifiers_;
386};
