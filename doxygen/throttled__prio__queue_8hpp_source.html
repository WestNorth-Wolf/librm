45 kEdf,

56template <typename T, usize MaxQueueSize, SchedulingPolicy Policy = SchedulingPolicy::kPriorityFifo>
57class ThrottledPrioQueue {
58 public:
59 using clock = std::chrono::steady_clock;
60 using time_point = clock::time_point;
61 using duration = clock::duration;
62
63 struct QueueItem {
64 T payload;
65 u8 priority;
66 time_point deadline;
67 usize enqueue_seq{0};
68
69 bool operator<(const QueueItem& other) const {
70 // 使用有符号差值比较序号，正确处理 enqueue_seq 上溢回绕
71 // 只要队列中同时存在的最大序号跨度 < usize 范围的一半，此方法即可正确工作
72 using signed_seq = std::make_signed_t<usize>;
73 const auto seq_after = [](usize a, usize b) -> bool { return static_cast<signed_seq>(a - b) > 0; };
74
75 if constexpr (Policy == SchedulingPolicy::kFifo) {
76 // FIFO：序号越小（越早入队）越应先出队
77 // 大顶堆中返回 true 表示 other 优先，故序号较晚（seq_after）的返回 true
78 return seq_after(enqueue_seq, other.enqueue_seq);
79 } else if constexpr (Policy == SchedulingPolicy::kPriorityFifo) {
80 // 优先级 + FIFO：优先级高的先出；同优先级时序号小的（早入队的）先出
81 if (priority == other.priority) {
82 return seq_after(enqueue_seq, other.enqueue_seq);
83 }
84 return priority < other.priority;
85 } else {
86 // EDF：优先级数值大的先出；同优先级时 deadline 早的先出
87 if (priority == other.priority) {
88 return deadline > other.deadline;
89 }
90 return priority < other.priority;
91 }
92 }
93 };
94
98 explicit ThrottledPrioQueue(double tx_frequency_hz)
99 : interval_(std::chrono::duration_cast<duration>(std::chrono::duration<double>(1.0 / tx_frequency_hz))) {}
100
108 bool Push(const T& payload, u8 priority, time_point deadline) {
109 if (queue_.full()) {
110 return false;
111 }
112 queue_.push({payload, priority, deadline, enqueue_seq_++});
113 return true;
114 }
115
120 std::optional<T> Process() { return Process(clock::now()); }
121
127 std::optional<T> Process(time_point now) {
128 // 1. 清理已超时的过期消息，精确计数
129 while (!queue_.empty()) {
130 if (now >= queue_.top().deadline) {
131 queue_.pop();
132 ++expired_count_;
133 } else {
134 break;
135 }
136 }
137
138 // 2. 检查队列是否为空或是否未到限频间隔
139 if (queue_.empty() || (now - last_process_time_ < interval_)) {
140 return std::nullopt;
141 }
142
143 // 3. 提取消息并更新状态
144 T payload = queue_.top().payload;
145 queue_.pop();
146 last_process_time_ = now;
147
148 return payload;
149 }
150
152 usize size() const { return queue_.size(); }
154 bool empty() const { return queue_.empty(); }
156 usize expired_count() const { return expired_count_; }
157 void Clear() {
158 while (!queue_.empty()) {
159 queue_.pop();
160 }
161 }
162
163 private:
164 etl::priority_queue<QueueItem, MaxQueueSize> queue_;
165 duration interval_;
166 time_point last_process_time_{};
167 usize expired_count_{0};
168 usize enqueue_seq_{0};
169};
170} // namespace rm::modules
