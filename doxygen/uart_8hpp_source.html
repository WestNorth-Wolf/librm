78constexpr usize kMaxUartInstances = 8;
79
80struct UartEntry {
81 UART_HandleTypeDef *handle{nullptr};
82 UartBase *obj{nullptr};
83};
84
85// inline 变量：C++17，所有 TU 共享同一对象，无 ODR 问题
86inline std::array<UartEntry, kMaxUartInstances> g_uart_registry{};
87
88inline void UartRegister(UART_HandleTypeDef *h, UartBase *obj) {
89 for (auto &e : g_uart_registry) {
90 if (e.handle == nullptr) {
91 e = {h, obj};
92 return;
93 }
94 }
95 Throw(std::runtime_error("uart_register: registry full"));
96}
97
98inline void UartUnregister(UART_HandleTypeDef *h) {
99 for (auto &e : g_uart_registry) {
100 if (e.handle == h) {
101 e = {nullptr, nullptr};
102 return;
103 }
104 }
105}
106
107inline UartBase *UartFind(UART_HandleTypeDef *h) {
108 for (auto &e : g_uart_registry) {
109 if (e.handle == h) return e.obj;
110 }
111 return nullptr;
112}

129template <usize RxBufSize>
130class Uart : public SyncSerialInterface,
131 public SerialInterface,
132 public AsyncTxSyncRxSerialInterface,
133 public AsyncSerialInterface,
134 private UartBase {
135 public:
136 explicit Uart(UART_HandleTypeDef &huart, bool async_tx_use_dma = false, bool async_rx_use_dma = false)
137 : huart_(&huart), async_tx_use_dma_(async_tx_use_dma), async_rx_use_dma_(async_rx_use_dma) {}
138
139 ~Uart() override {
140 Uart<RxBufSize>::Stop();
141 detail::UartUnregister(huart_);
142 }
143
144 // /// @name SyncWritable
145 void Write(const u8 *data, usize size, u32 timeout_ms) override {
146 LIBRM_STM32_HAL_ASSERT(HAL_UART_Transmit(huart_, const_cast<u8 *>(data), size, timeout_ms));
147 }
148
149 // /// @name AsyncWritable
150 void WriteAsync(const u8 *data, usize size, std::function<void()> on_done) override {
151 tx_done_callback_ = std::move(on_done);
152#ifdef HAL_DMA_MODULE_ENABLED
153 if (async_tx_use_dma_) {
154 LIBRM_STM32_HAL_ASSERT(HAL_UART_Transmit_DMA(huart_, const_cast<u8 *>(data), size));
155 } else
156#endif
157 {
158 LIBRM_STM32_HAL_ASSERT(HAL_UART_Transmit_IT(huart_, const_cast<u8 *>(data), size));
159 }
160 }
161
162 bool IsTxBusy() const override { return huart_->gState != HAL_UART_STATE_READY; }
163
164 // /// @name SyncReadable
165 usize Read(u8 *buf, usize size, u32 timeout_ms) override {
166 const HAL_StatusTypeDef s = HAL_UART_Receive(huart_, buf, static_cast<u16>(size), timeout_ms);
167 return (s == HAL_OK) ? size : 0u;
168 }
169
170 // /// @name AsyncReadable
171 void AttachRxCallback(SerialRxCallbackFunction callback) override { rx_callbacks_.emplace_back(std::move(callback)); }
172
173 void Start() override {
174 if (is_receiving_) {
175 return;
176 }
177#ifdef HAL_DMA_MODULE_ENABLED
178 if (async_rx_use_dma_ && huart_->hdmarx == nullptr) {
179 Throw(std::runtime_error("async_rx_use_dma_ selected but hdmarx is nullptr"));
180 }
181#endif
182 detail::UartRegister(huart_, this);
183
184 // 注册 RX 完成回调（无捕获 lambda → 函数指针）
185 LIBRM_STM32_HAL_ASSERT(HAL_UART_RegisterRxEventCallback(huart_, [](UART_HandleTypeDef *h, u16 len) {
186 if (auto *obj = detail::UartFind(h)) {
187 obj->HalRxCpltCallback(len);
188 }
189 }));
190
191 // 注册错误回调
192 LIBRM_STM32_HAL_ASSERT(HAL_UART_RegisterCallback(huart_, HAL_UART_ERROR_CB_ID, [](UART_HandleTypeDef *h) {
193 if (auto *obj = detail::UartFind(h)) {
194 obj->HalErrorCallback();
195 }
196 }));
197
198 // 注册 TX 完成回调（供 WriteAsync 使用）
199 LIBRM_STM32_HAL_ASSERT(HAL_UART_RegisterCallback(huart_, HAL_UART_TX_COMPLETE_CB_ID, [](UART_HandleTypeDef *h) {
200 if (auto *obj = detail::UartFind(h)) {
201 obj->HalTxCpltCallback();
202 }
203 }));
204
205 is_receiving_ = true;
206 RestartRx();
207 }
208
209 void Stop() override {
210 if (!is_receiving_) {
211 return;
212 }
213 is_receiving_ = false;
214 HAL_UART_AbortReceive(huart_);
215 }
216
217 private:
218 void RestartRx() {
219 auto &buf = rx_buf_[!buffer_selector_];
220#ifdef HAL_DMA_MODULE_ENABLED
221 if (async_rx_use_dma_) {
222 LIBRM_STM32_HAL_ASSERT(HAL_UARTEx_ReceiveToIdle_DMA(huart_, buf.data(), buf.size()));
223 __HAL_DMA_DISABLE_IT(huart_->hdmarx, DMA_IT_HT);
224 } else
225#endif
226 {
227 LIBRM_STM32_HAL_ASSERT(HAL_UARTEx_ReceiveToIdle_IT(huart_, buf.data(), buf.size()));
228 }
229 }
230
231 void HalRxCpltCallback(u16 rx_len) override {
232 if (!is_receiving_) {
233 return;
234 }
235 RestartRx(); // 先重启接收，最小化数据丢失窗口
236 etl::span<const u8> received{rx_buf_[buffer_selector_].data(), rx_len};
237 for (auto &cb : rx_callbacks_) {
238 if (cb) {
239 cb(received);
240 }
241 }
242 buffer_selector_ = !buffer_selector_;
243 }
244
245 void HalTxCpltCallback() override {
246 if (!tx_done_callback_) {
247 return;
248 }
249 auto cb = std::move(tx_done_callback_);
250 tx_done_callback_ = nullptr;
251 cb();
252 }
253
254 void HalErrorCallback() override {
255 if (!is_receiving_) {
256 return;
257 }
258 is_receiving_ = false;
259 RestartRx();
260 is_receiving_ = true;
261 }
262
263 etl::vector<SerialRxCallbackFunction, 10> rx_callbacks_;
264 std::function<void()> tx_done_callback_{nullptr};
265 UART_HandleTypeDef *huart_;
266 bool async_tx_use_dma_;
267 bool async_rx_use_dma_;
268 std::array<u8, RxBufSize> rx_buf_[2]{};
269 bool buffer_selector_{false};
270 bool is_receiving_{false};
271};
