74 void RxCallback(const hal::CanFrame *msg) override {
75 // 转换为HIPNUC SDK的帧格式
76 hipnuc_can_frame_t frame;
77 frame.can_id = msg->rx_std_id;
78 frame.can_dlc = msg->dlc;
79 for (u8 i = 0; i < msg->dlc && i < 8; ++i) {
80 frame.data[i] = msg->data[i];
81 }
82 frame.hw_ts_us = 0; // 时间戳暂不使用
83
84 // 使用HIPNUC SDK解析帧
85 int msg_type = canopen_parse_frame(&frame, &sensor_data_);
86
87 // 根据消息类型更新对应的数据
88 if (msg_type != CAN_MSG_ERROR && msg_type != CAN_MSG_UNKNOWN) {
89 ReportStatus(kOk);
90 last_msg_type_ = msg_type;
91
92 // 更新对应的数据字段
93 switch (msg_type) {
94 case CAN_MSG_ACCEL:
95 acc_[0] = sensor_data_.acc_x;
96 acc_[1] = sensor_data_.acc_y;
97 acc_[2] = sensor_data_.acc_z;
98 break;
99 case CAN_MSG_GYRO:
100 gyro_[0] = sensor_data_.gyr_x;
101 gyro_[1] = sensor_data_.gyr_y;
102 gyro_[2] = sensor_data_.gyr_z;
103 break;
104 case CAN_MSG_EULER:
105 roll_ = sensor_data_.roll;
106 pitch_ = sensor_data_.pitch;
107 yaw_ = sensor_data_.imu_yaw;
108 break;
109 case CAN_MSG_QUAT:
110 quat_[0] = sensor_data_.quat_w;
111 quat_[1] = sensor_data_.quat_x;
112 quat_[2] = sensor_data_.quat_y;
113 quat_[3] = sensor_data_.quat_z;
114 break;
115 case CAN_MSG_PRESSURE:
116 pressure_ = sensor_data_.pressure;
117 break;
118 case CAN_MSG_INCLI:
119 incli_x_ = sensor_data_.incli_x;
120 incli_y_ = sensor_data_.incli_y;
121 break;
122 default:
123 break;
124 }
125 }
126 }
