107 bool Write(u16 id, const u8 *data, usize size, u8 priority, time_point deadline) {
108 TxRequest req;
109 req.id = id;
110 req.size = (size <= MaxDataSize) ? size : MaxDataSize;
111 std::copy_n(data, req.size, req.data.begin());
112 ++stat_window_data_.enqueue_count;
113 if (!queue_.Push(req, priority, deadline)) {
114 ++stat_window_data_.drop_full_count;
115 return false;
116 }
117 return true;
118 }
119
125 bool Process() { return Process(clock::now()); }
126
132 bool Process(time_point now) {
133 // 采样队列深度
134 const usize depth = queue_.size();
135 depth_avg_.add(static_cast<f32>(depth));
136 if (depth > stat_window_data_.peak_depth) {
137 stat_window_data_.peak_depth = depth;
138 }
139
140 auto result = queue_.Process(now);
141 const bool sent = result.has_value();
142 if (sent) {
143 const auto &req = result.value();
144 Base::Write(req.id, req.data.data(), req.size);
145 ++stat_window_data_.tx_count;
146 }
147
148 // 检查是否到了 1 秒刷新窗口
149 if (now - stats_window_start_ >= std::chrono::seconds(1)) {
150 const f32 elapsed_s = std::chrono::duration<f32>(now - stats_window_start_).count();
151
152 const usize expired_delta = queue_.expired_count() - stat_window_data_.expired_baseline;
153
154 // clang-format off
155 TxStats snap;
156 snap.tx_fps = static_cast<f32>(stat_window_data_.tx_count) / elapsed_s;
157 snap.enqueue_fps = static_cast<f32>(stat_window_data_.enqueue_count) / elapsed_s;
158 snap.drop_full_fps = static_cast<f32>(stat_window_data_.drop_full_count) / elapsed_s;
159 snap.drop_expired_fps = static_cast<f32>(expired_delta) / elapsed_s;
160 snap.drop_total_fps = snap.drop_full_fps + snap.drop_expired_fps;
161 snap.peak_queue_depth = stat_window_data_.peak_depth;
162 snap.avg_queue_depth = depth_avg_.value();
163 stats_snapshot_ = snap;
164
165 // 重置窗口
166 stats_window_start_ = now;
167 stat_window_data_.tx_count = 0;
168 stat_window_data_.enqueue_count = 0;
169 stat_window_data_.drop_full_count = 0;
170 stat_window_data_.expired_baseline = queue_.expired_count();
171 stat_window_data_.peak_depth = 0;
172 depth_avg_.clear(0.0f);
173 // clang-format on
174 }
175
176 return sent;
177 }
