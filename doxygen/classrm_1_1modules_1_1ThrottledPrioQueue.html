template<typename T, usize MaxQueueSize, SchedulingPolicy Policy = SchedulingPolicy::kPriorityFifo>
class rm::modules::ThrottledPrioQueue< T, MaxQueueSize, Policy >

模板参数
T负载数据类型 (如CanFrame)

Policy调度策略，默认 kFifo
